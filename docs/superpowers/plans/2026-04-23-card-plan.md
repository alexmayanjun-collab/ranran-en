- [ ] **Step 2: 添加渲染函数**

```javascript
function renderCard() {
  const data = englishData[currentDate];
  if (!data) {
    document.getElementById('app').innerHTML = 
      '<div style="text-align:center;padding:50px;color:#666;">暂无数据</div>';
    return;
  }

  const config = sceneConfig[data.scene] || sceneConfig["早上见面"];
  
  const card = document.createElement('div');
  card.className = 'scene-card';
  card.style.background = config.gradient;
  
  // 生成气泡 HTML
  const bubblesHtml = data.dialogs.map((d, i) => `
    <div class="dialog-bubble ${i % 2 === 0 ? 'parent' : 'child'}" style="animation-delay: ${0.2 + i * 0.15}s">
      <div class="english">${d.english}</div>
      <div class="chinese">${d.chinese}</div>
    </div>
  `).join('');
  
  card.innerHTML = `
    <div class="card-icon">${config.icon}</div>
    <div class="scene-title">${config.title}</div>
    <div class="dialog-list">${bubblesHtml}</div>
    <div class="check-area">
      <div class="check-item" onclick="toggleCheck(this)">
        <div class="check-icon"></div>
        <span>跟读完成</span>
      </div>
      <div class="check-item" onclick="toggleCheck(this)">
        <div class="check-icon"></div>
        <span>场景应用</span>
      </div>
    </div>
  `;
  
  document.getElementById('app').appendChild(card);
  loadCheckStatus();
}
```

- [ ] **Step 3: 添加打卡功能**

```javascript
function toggleCheck(el) {
  el.classList.toggle('checked');
  el.querySelector('.check-icon').innerHTML = 
    el.classList.contains('checked') ? '✓' : '';
  saveCheckStatus();
}

function saveCheckStatus() {
  const checks = document.querySelectorAll('.check-item');
  const status = Array.from(checks).map(c => c.classList.contains('checked'));
  localStorage.setItem('english-check-' + currentDate, JSON.stringify(status));
}

function loadCheckStatus() {
  const saved = localStorage.getItem('english-check-' + currentDate);
  if (saved) {
    const status = JSON.parse(saved);
    const checks = document.querySelectorAll('.check-item');
    checks.forEach((c, i) => {
      if (status[i]) {
        c.classList.add('checked');
        c.querySelector('.check-icon').innerHTML = '✓';
      }
    });
  }
}
```
